\begin{definition}[punkt Gergonne'a] % Guzicki s. 134
\index{punkt!Gergonne'a}%
\label{punkt_gergonne}

	Wtedy czewiany $AP$, $BQ$ i $CR$ są współpękowe.
\end{definition}
